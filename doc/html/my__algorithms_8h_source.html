123  template<class T>
124  static int
125  partition(T *array, int lowestIndexPossible, int highestIndexPossible,
126  int indexOfElementToServeAsPivot = 0) {
127 
128  /*
129  * set initial-`pivotIndex` to be the `first` `element` in the array,
130  * thus set the `nonPivotIndex` to be the `last` in the array.
131  * This is set arbitrarily, but is not necessarily important.
132  */
133  int pivotIndex = lowestIndexPossible;
134  if (indexOfElementToServeAsPivot != 0) {
135 
136  /*
137  * if there is a request to know the specific `pivot-index`
138  * of a specific-given `index`, set the `element` of this
139  * specific-given `index` to be the initial-`pivot-element`.
140  */
141  swap<T>(array, pivotIndex, indexOfElementToServeAsPivot);
142  }
143  int nonPivotIndex = highestIndexPossible;
144 
145  while (true) {
146  if (pivotIndex < nonPivotIndex) {
147  if (array[pivotIndex] > array[nonPivotIndex]) {
148  swap<T>(array, pivotIndex, nonPivotIndex);
149  swap<int>(pivotIndex, nonPivotIndex);
150  ++nonPivotIndex;
151  } else { --nonPivotIndex; }
152  } else if (nonPivotIndex < pivotIndex) {
153  if (array[nonPivotIndex] > array[pivotIndex]) {
154  swap<T>(array, pivotIndex, nonPivotIndex);
155  swap<int>(pivotIndex, nonPivotIndex);
156  --nonPivotIndex;
157  } else { ++nonPivotIndex; }
158  } else if (nonPivotIndex == pivotIndex) {
159  return pivotIndex;
160  }
161  }
162  }

310  whileOfBubbleSort_division_TheIthElementInItsSize(T *array, int size,
311  int &divisionCounter,
312  int &divisionInteger,

315  divisionInteger = (size / (divisionCounter * DIVISION));
316  if (divisionInteger < 1) {
