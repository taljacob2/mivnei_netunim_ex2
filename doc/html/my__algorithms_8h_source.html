342  static int partition(T *array, int lowestIndexPossible,
343  int highestIndexPossible,
344  int indexOfElementToServeAsPivot = 0) {
345 
346  /*
347  * set initial-`pivotIndex` to be the `first` `element` in the array,
348  * thus set the `nonPivotIndex` to be the `last` in the array.
349  * This is set arbitrarily, but is not necessarily important.
350  */
351  int pivotIndex = lowestIndexPossible;
352  if (indexOfElementToServeAsPivot != 0) {
353 
354  /*
355  * if there is a request to know the specific `pivot-index`
356  * of a specific-given `index`, set the `element` of this
357  * specific-given `index` to be the initial-`pivot-element`.
358  */
359  swap<T>(array, pivotIndex, indexOfElementToServeAsPivot);
360  }
361  int nonPivotIndex = highestIndexPossible;
362 
363  while (true) {
364  if (pivotIndex < nonPivotIndex) {
365  if (array[pivotIndex] > array[nonPivotIndex]) {
366  swap<T>(array, pivotIndex, nonPivotIndex);
367  swap<int>(pivotIndex, nonPivotIndex);
368  ++nonPivotIndex;
369  } else {
370  --nonPivotIndex;
371  }
372  } else if (nonPivotIndex < pivotIndex) {
373  if (array[nonPivotIndex] > array[pivotIndex]) {
374  swap<T>(array, pivotIndex, nonPivotIndex);
375  swap<int>(pivotIndex, nonPivotIndex);
376  --nonPivotIndex;
377  } else {
378  ++nonPivotIndex;
379  }
380  } else if (nonPivotIndex == pivotIndex) {
381  return pivotIndex;
382  }
