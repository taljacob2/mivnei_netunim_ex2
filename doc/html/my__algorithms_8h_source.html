179  static int partition(T *array, int lowestIndexPossible,
180  int highestIndexPossible,
181  int indexOfElementToServeAsPivot = 0) {
182 
183  /*
184  * set initial-`pivotIndex` to be the `first` `element` in the array,
185  * thus set the `nonPivotIndex` to be the `last` in the array.
186  * This is set arbitrarily, but is not necessarily important.
187  */
188  int pivotIndex = lowestIndexPossible;
189  if (indexOfElementToServeAsPivot != 0) {
190 
191  /*
192  * if there is a request to know the specific `pivot-index`
193  * of a specific-given `index`, set the `element` of this
194  * specific-given `index` to be the initial-`pivot-element`.
195  */
196  swap<T>(array, pivotIndex, indexOfElementToServeAsPivot);
197  }
198  int nonPivotIndex = highestIndexPossible;
199 
200  while (true) {
201  if (pivotIndex < nonPivotIndex) {
202  if (array[pivotIndex] > array[nonPivotIndex]) {
203  swap<T>(array, pivotIndex, nonPivotIndex);
204  swap<int>(pivotIndex, nonPivotIndex);
205  ++nonPivotIndex;
206  } else {
207  --nonPivotIndex;
208  }
209  } else if (nonPivotIndex < pivotIndex) {
210  if (array[nonPivotIndex] > array[pivotIndex]) {
211  swap<T>(array, pivotIndex, nonPivotIndex);
212  swap<int>(pivotIndex, nonPivotIndex);
213  --nonPivotIndex;
214  } else {
215  ++nonPivotIndex;
216  }
217  } else if (nonPivotIndex == pivotIndex) {
218  return pivotIndex;
219  }
