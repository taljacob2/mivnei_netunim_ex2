344  static long int partition(T *array, long int lowestIndexPossible,
345  long int highestIndexPossible,
346  long int indexOfElementToServeAsPivot = 0) {
347 
348  /*
349  * set initial-`pivotIndex` to be the `first` `element` in the array,
350  * thus set the `nonPivotIndex` to be the `last` in the array.
351  * This is set arbitrarily, but is not necessarily important.
352  */
353  long int pivotIndex = lowestIndexPossible;
354  if (indexOfElementToServeAsPivot != 0) {
355 
356  /*
357  * if there is a request to know the specific `pivot-index`
358  * of a specific-given `index`, set the `element` of this
359  * specific-given `index` to be the initial-`pivot-element`.
360  */
361  swap<T>(array, pivotIndex, indexOfElementToServeAsPivot);
362  }
363  long int nonPivotIndex = highestIndexPossible;
364 
365  while (true) {
366  if (pivotIndex < nonPivotIndex) {
367  if (array[pivotIndex] > array[nonPivotIndex]) {
368  swap<T>(array, pivotIndex, nonPivotIndex);
369  swap<long int>(pivotIndex, nonPivotIndex);
370  ++nonPivotIndex;
371  } else {
372  --nonPivotIndex;
373  }
374  } else if (nonPivotIndex < pivotIndex) {
375  if (array[nonPivotIndex] > array[pivotIndex]) {
376  swap<T>(array, pivotIndex, nonPivotIndex);
377  swap<long int>(pivotIndex, nonPivotIndex);
378  --nonPivotIndex;
379  } else {
380  ++nonPivotIndex;
381  }
382  } else if (nonPivotIndex == pivotIndex) {
383  return pivotIndex;
384  }
