337  void fixHeapWhile(int currentIndex) {
338 
339  /* array[currentIndex] is not `nullptr`. Thus, comparable. */
340  while ((0 <= currentIndex) && (currentIndex < (logicalSize / 2))) {
341 
342  /* Get the index that points to the `minimal` element. */
343  int indexOfMinimalChildOfCurrentRoot =
344  my_algorithms::min(array, logicalSize, currentIndex * 2 + 1,
345  currentIndex * 2 + 2);
346  if (array[indexOfMinimalChildOfCurrentRoot] != nullptr) {
347 
348  /*
349  * There is a living entry.
350  * Compare by keys.
351  * `swap` the elements if needed, to maintain the validity of
352  * the heap as a `Minimum-Heap`.
353  */
354  if (*array[currentIndex] >
355  *array[indexOfMinimalChildOfCurrentRoot]) {
356  my_algorithms::swap(array, currentIndex,
357  indexOfMinimalChildOfCurrentRoot);
358 
359  /*
360  * Set the updated iterator index to the replaced index.
361  * Note: this enlarges the index.
362  */
363  currentIndex = indexOfMinimalChildOfCurrentRoot;
364  } else {
365  /*
366  * Second stop condition:
367  * the parent is not smaller than its child. break.
368  */
369  break;
370  }
371  } else {
372 
373  /*
374  * There is no entry to compare with.
375  * Thus, this `node` does not have children,
376  * and is actually a leaf.
377  */
378  break;
379  }
380  }
