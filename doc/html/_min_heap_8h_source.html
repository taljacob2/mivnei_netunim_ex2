249  void fixHeapWhile(int currentIndex) {
250 
251  /* array[currentIndex] is not `nullptr`. Thus, comparable. */
252  while ((0 <= currentIndex) && (currentIndex < (logicalSize / 2))) {
253 
254  /* Get the index that points to the `minimal` element. */
255  int indexOfMinimalChildOfCurrentRoot =
256  my_algorithms::min(array, logicalSize, currentIndex * 2 + 1,
257  currentIndex * 2 + 2);
258  if (array[indexOfMinimalChildOfCurrentRoot] != nullptr) {
259 
260  /*
261  * There is a living entry.
262  * Compare by keys.
263  * `swap` the elements if needed, to maintain the validity of
264  * the heap as a `Minimum-Heap`.
265  */
266  if (*array[currentIndex] >
267  *array[indexOfMinimalChildOfCurrentRoot]) {
268  my_algorithms::swap(array, currentIndex,
269  indexOfMinimalChildOfCurrentRoot);
270

272  * Set the updated iterator index to the replaced index.
